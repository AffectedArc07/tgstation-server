82  async Task<MonitorAction> HandleMonitorWakeup(MonitorActivationReason reason, CancellationToken cancellationToken)
83  {
84  switch (reason)
85  {
86  case MonitorActivationReason.ActiveServerCrashed:
87  string exitWord = server.TerminationWasRequested ? "exited" : "crashed";
88  if (server.RebootState == Watchdog.RebootState.Shutdown)
89  {
90  // the time for graceful shutdown is now
91  await Chat.SendWatchdogMessage(String.Format(CultureInfo.InvariantCulture, "Server {0}! Exiting due to graceful termination request...", exitWord), cancellationToken).ConfigureAwait(false);
92  DisposeAndNullControllers();
93  return MonitorAction.Exit;
94  }
95 
96  await Chat.SendWatchdogMessage(String.Format(CultureInfo.InvariantCulture, "Server {0}! Rebooting...", exitWord), cancellationToken).ConfigureAwait(false);
97  return MonitorAction.Restart;
98  case MonitorActivationReason.ActiveServerRebooted:
99  var rebootState = server.RebootState;
100  server.ResetRebootState();
101 
102  switch (rebootState)
103  {
104  case Watchdog.RebootState.Normal:
105  bool dmbUpdatePending = ActiveLaunchParameters != LastLaunchParameters;
106  return dmbUpdatePending ? MonitorAction.Restart : MonitorAction.Continue;
107  case Watchdog.RebootState.Restart:
108  return MonitorAction.Restart;
109  case Watchdog.RebootState.Shutdown:
110  // graceful shutdown time
111  await Chat.SendWatchdogMessage("Active server rebooted! Exiting due to graceful termination request...", cancellationToken).ConfigureAwait(false);
112  DisposeAndNullControllers();
113  return MonitorAction.Exit;
114  default:
115  throw new InvalidOperationException($"Invalid reboot state: {rebootState}");
116  }
117 
118  case MonitorActivationReason.ActiveLaunchParametersUpdated:
119  case MonitorActivationReason.NewDmbAvailable:
120  await server.SetRebootState(Watchdog.RebootState.Restart, cancellationToken).ConfigureAwait(false);
121  return MonitorAction.Continue;
122  case MonitorActivationReason.InactiveServerCrashed:
123  case MonitorActivationReason.InactiveServerRebooted:
124  case MonitorActivationReason.InactiveServerStartupComplete:
125  throw new NotSupportedException($"Unsupported activation reason: {reason}");
126  default:
127  throw new InvalidOperationException($"Invalid activation reason: {reason}");
128  }

239  // always start out with continue

267  {

269  // multiple things may have happened, handle them one at a time
270  for (var moreActivationsToProcess = true; moreActivationsToProcess && (nextAction == MonitorAction.Continue || nextAction == MonitorAction.Skip);)
271  {

279  if (nextAction == MonitorAction.Skip)
